<script>
    /* --- LÓGICA MATEMÁTICA (Portado del JSX) --- */

    let chartInstance = null;

    // Colores
    const C = {
        sistema: "#F58220", // Naranja Elecnor
        simple: "#38bdf8",  // Azul claro
        paralelo: "#34d399",// Verde
        serie: "#f472b6"    // Rosa
    };

    function getVal(id) {
        return parseFloat(document.getElementById(id).value) || 0;
    }

    function calcK_sistema(H_f, Q_dis) {
        if (Q_dis <= 0) return 0;
        return H_f / (Q_dis * Q_dis);
    }

    function calcK_bomba(H0, H_nom, Q_nom) {
        if (Q_nom <= 0 || H0 <= H_nom) return 0.001; // Evitar división por cero o K negativa
        return (H0 - H_nom) / (Q_nom * Q_nom);
    }

    // Funciones de intersección exactas (Resolución de ecuaciones cuadráticas)
    function intersectSimple(H_est, k_sys, H0, k_pump, Qmax) {
        const denom = k_sys + k_pump;
        if (denom <= 0) return null;
        const Q2 = (H0 - H_est) / denom;
        if (Q2 <= 0) return null;
        const Q = Math.sqrt(Q2);
        if (Q > Qmax * 1.5) return null; // Fuera de rango razonable
        return { Q: Q, H: H_est + k_sys * Q2 };
    }

    function intersectParallel(H_est, k_sys, H0, k_pump, Qmax) {
        // En paralelo, Q_total = 2 * Q_bomba_individual.
        // H_sis = H_est + k_sys * Q^2
        // H_bombas = H0 - k_pump * (Q/2)^2 = H0 - (k_pump/4) * Q^2
        const denom = k_sys + k_pump / 4;
        if (denom <= 0) return null;
        const Q2 = (H0 - H_est) / denom;
        if (Q2 <= 0) return null;
        const Q = Math.sqrt(Q2);
        return { Q: Q, H: H_est + k_sys * Q2 };
    }

    function intersectSeries(H_est, k_sys, H0, k_pump, Qmax) {
        // En serie, H_total = 2 * H_bomba. Q es el mismo.
        // H_bombas = 2 * (H0 - k_pump * Q^2) = 2*H0 - 2*k_pump*Q^2
        const denom = k_sys + 2 * k_pump;
        if (denom <= 0) return null;
        const Q2 = (2 * H0 - H_est) / denom;
        if (Q2 <= 0) return null;
        const Q = Math.sqrt(Q2);
        return { Q: Q, H: H_est + k_sys * Q2 };
    }

    function updateChart() {
        // 1. Obtener inputs
        const p = {
            H_g: getVal('H_g'),
            P_res: getVal('P_res'),
            H_f: getVal('H_f'),
            Q_dis: getVal('Q_dis'),
            H0: getVal('H0'),
            H_nom: getVal('H_nom'),
            Q_nom: getVal('Q_nom'),
            Q_max: getVal('Q_max')
        };

        const show = {
            sys: document.getElementById('show_sys').checked,
            simple: document.getElementById('show_simple').checked,
            parallel: document.getElementById('show_parallel').checked,
            series: document.getElementById('show_series').checked,
        };

        // 2. Cálculos Constantes
        const H_est = p.H_g + p.P_res;
        const k_sys = calcK_sistema(p.H_f, p.Q_dis);
        const k_pump = calcK_bomba(p.H0, p.H_nom, p.Q_nom);
        
        // Calcular puntos de intersección
        const ptS = intersectSimple(H_est, k_sys, p.H0, k_pump, p.Q_max);
        const ptP = intersectParallel(H_est, k_sys, p.H0, k_pump, p.Q_max);
        const ptSer = intersectSeries(H_est, k_sys, p.H0, k_pump, p.Q_max);

        // 3. Generar Datos para Gráfica (muestreo)
        const steps = 100; // Resolución
        const labels = []; // Eje X
        const d_sys = [], d_simple = [], d_para = [], d_serie = [];

        // Definimos el rango X. Usamos Q_max del input.
        for (let i = 0; i <= steps; i++) {
            const Q = (i / steps) * p.Q_max;
            labels.push(Q.toFixed(1));

            // Sistema: H = H_est + k*Q^2
            if (show.sys) {
                const H = H_est + k_sys * (Q * Q);
                d_sys.push(H);
            }

            // Simple: H = H0 - k*Q^2
            if (show.simple) {
                const H = p.H0 - k_pump * (Q * Q);
                d_simple.push(H > 0 ? H : null);
            }

            // Paralelo: H = H0 - k*(Q/2)^2
            if (show.parallel) {
                const H = p.H0 - k_pump * ((Q/2) * (Q/2));
                d_para.push(H > 0 ? H : null);
            }

            // Serie: H = 2*(H0 - k*Q^2)
            if (show.series) {
                const H_single = p.H0 - k_pump * (Q * Q);
                const H = H_single * 2;
                d_serie.push(H > 0 ? H : null);
            }
        }

        // 4. Actualizar Resultados HTML (Badges)
        const resContainer = document.getElementById('resultsContainer');
        resContainer.innerHTML = ''; // Limpiar

        function createBadge(title, pt, color, extra='') {
            if(!pt) return;
            const html = `
                <div class="badge" style="border-color:${color}">
                    <h4 style="color:${color}">${title}</h4>
                    <div class="value">Q: ${pt.Q.toFixed(2)} m³/h</div>
                    <div class="value">H: ${pt.H.toFixed(2)} m.c.a.</div>
                    ${extra ? `<div style="font-size:0.75em;color:#666;margin-top:3px">${extra}</div>` : ''}
                </div>`;
            resContainer.innerHTML += html;
        }

        // Diseño (Sistema)
        if(show.sys) {
            // Calculamos H a Q_dis
            const H_at_design = H_est + k_sys * Math.pow(p.Q_dis, 2);
            resContainer.innerHTML += `
                <div class="badge sys">
                    <h4 style="color:${C.sistema}">Punto Diseño (Requerido)</h4>
                    <div class="value">Q: ${p.Q_dis.toFixed(1)} m³/h</div>
                    <div class="value">H: ${H_at_design.toFixed(2)} m.c.a.</div>
                    <div style="font-size:0.75em;color:#666;margin-top:3px">H_est: ${H_est.toFixed(2)}m</div>
                </div>`;
        }
        
        if (show.simple) createBadge("Intersección 1 Bomba", ptS, C.simple);
        if (show.parallel) createBadge("Intersección Paralelo", ptP, C.paralelo, ptP ? `Cada bomba: ${(ptP.Q/2).toFixed(1)} m³/h` : '');
        if (show.series) createBadge("Intersección Serie", ptSer, C.serie);


        // 5. Configurar Dataset de Chart.js
        const datasets = [];

        // Puntos de trabajo (Scatter)
        const pointRadius = 6;
        const pointHover = 8;
        
        if (show.sys && p.Q_dis <= p.Q_max) {
             datasets.push({
                label: 'Punto Diseño',
                data: [{x: p.Q_dis, y: H_est + k_sys * Math.pow(p.Q_dis, 2)}],
                borderColor: C.sistema, backgroundColor: 'white',
                borderWidth: 2, pointRadius: 5, pointStyle: 'circle', type: 'scatter'
            });
        }
        if (show.simple && ptS) {
            datasets.push({
                label: 'Op 1 Bomba',
                data: [{x: ptS.Q, y: ptS.H}],
                backgroundColor: C.simple, borderColor: '#fff',
                pointRadius: pointRadius, pointHoverRadius: pointHover, type: 'scatter'
            });
        }
        if (show.parallel && ptP) {
            datasets.push({
                label: 'Op Paralelo',
                data: [{x: ptP.Q, y: ptP.H}],
                backgroundColor: C.paralelo, borderColor: '#fff',
                pointRadius: pointRadius, pointHoverRadius: pointHover, type: 'scatter'
            });
        }
        if (show.series && ptSer) {
            datasets.push({
                label: 'Op Serie',
                data: [{x: ptSer.Q, y: ptSer.H}],
                backgroundColor: C.serie, borderColor: '#fff',
                pointRadius: pointRadius, pointHoverRadius: pointHover, type: 'scatter'
            });
        }

        // Líneas
        if (show.sys) {
            datasets.push({
                label: 'Curva Sistema', data: d_sys,
                borderColor: C.sistema, borderWidth: 2, borderDash: [5, 5],
                pointRadius: 0, fill: false, tension: 0.4
            });
        }
        if (show.simple) {
            datasets.push({
                label: '1 Bomba', data: d_simple,
                borderColor: C.simple, borderWidth: 2,
                pointRadius: 0, fill: false, tension: 0.4
            });
        }
        if (show.parallel) {
            datasets.push({
                label: '2 Paralelo', data: d_para,
                borderColor: C.paralelo, borderWidth: 2,
                pointRadius: 0, fill: false, tension: 0.4
            });
        }
        if (show.series) {
            datasets.push({
                label: '2 Serie', data: d_serie,
                borderColor: C.serie, borderWidth: 2,
                pointRadius: 0, fill: false, tension: 0.4
            });
        }

        // 6. Renderizar Chart
        const ctx = document.getElementById('pumpChart').getContext('2d');
        
        if (chartInstance) {
            chartInstance.destroy();
        }

        chartInstance = new Chart(ctx, {
            type: 'line',
            data: {
                labels: labels,
                datasets: datasets
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                interaction: {
                    mode: 'index',
                    intersect: false,
                },
                plugins: {
                    legend: {
                        position: 'top',
                        labels: { usePointStyle: true, boxWidth: 8 }
                    },
                    tooltip: {
                        backgroundColor: 'rgba(0, 48, 135, 0.9)',
                        titleFont: { family: 'monospace' },
                        bodyFont: { family: 'monospace' },
                        callbacks: {
                            title: (ctx) => `Q = ${ctx[0].label} m³/h`,
                            label: (ctx) => {
                                if(ctx.dataset.type === 'scatter') return `${ctx.dataset.label}: ${ctx.raw.y.toFixed(2)} mca`;
                                return `${ctx.dataset.label}: ${ctx.raw.toFixed(2)} mca`;
                            }
                        }
                    }
                },
                scales: {
                    x: {
                        type: 'linear',
                        position: 'bottom',
                        title: { display: true, text: 'Caudal Q (m³/h)', color: '#64748b' },
                        grid: { color: '#f1f5f9' },
                        max: p.Q_max
                    },
                    y: {
                        title: { display: true, text: 'Altura H (m.c.a.)', color: '#64748b' },
                        grid: { color: '#f1f5f9' },
                        min: 0
                    }
                }
            }
        });
    }

    // Inicializar al cargar
    window.onload = updateChart;

</script>
