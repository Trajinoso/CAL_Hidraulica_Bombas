<script>
    /* ─── LÓGICA DE NEGOCIO (Matemáticas) ─── */

    let chartInstance = null;
    
    // Estado global de N
    let state = {
        n_paralelo: 2,
        n_serie: 2
    };

    // Colores
    const C = {
        sys: "#F58220",
        simple: "#38bdf8",
        par: "#10b981",
        ser: "#ec4899"
    };

    // Funciones Helper
    const getVal = (id) => parseFloat(document.getElementById(id).value) || 0;
    
    // Función para botones N
    function setN(type, val) {
        if(type === 'par') state.n_paralelo = val;
        if(type === 'ser') state.n_serie = val;

        // Actualizar UI botones
        document.querySelectorAll(`#btn-group-${type} .n-btn`).forEach(b => b.classList.remove('active'));
        // El botón clickeado no lo tengo directamente aquí, busco por texto o índice
        // Truco simple: seleccionar por orden (val-1)
        document.querySelectorAll(`#btn-group-${type} .n-btn`)[val-1].classList.add('active');

        updateChart();
    }

    // Cálculos de Coeficientes K
    function calcK_sistema(H_f, Q_dis) {
        if (Q_dis <= 0) return 0;
        return H_f / (Q_dis * Q_dis);
    }

    function calcK_bomba(H0, H_nom, Q_nom) {
        if (Q_nom <= 0 || H0 <= H_nom) return 0.001;
        return (H0 - H_nom) / (Q_nom * Q_nom);
    }

    /* ─── INTERSECCIONES ─── */
    
    // Intersección N bombas PARALELO con sistema
    // H_sis = H_est + k_sys * Q^2
    // H_bombas = H0 - k_pump * (Q/N)^2
    function intersectParN(H_est, k_sys, H0, k_pump, N, Qmax) {
        const denom = k_sys + k_pump / (N * N);
        if (denom <= 0) return null;
        const Q2 = (H0 - H_est) / denom;
        if (Q2 <= 0) return null;
        const Q = Math.sqrt(Q2);
        const H = H_est + k_sys * Q2;
        if (Q > Qmax * 1.5) return null; // Fuera de rango razonable
        return { Q, H };
    }

    // Intersección N bombas SERIE con sistema
    // H_sis = H_est + k_sys * Q^2
    // H_bombas = N * (H0 - k_pump * Q^2) = N*H0 - N*k_pump*Q^2
    function intersectSerN(H_est, k_sys, H0, k_pump, N, Qmax) {
        const denom = k_sys + N * k_pump;
        if (denom <= 0) return null;
        const Q2 = (N * H0 - H_est) / denom;
        if (Q2 <= 0) return null;
        const Q = Math.sqrt(Q2);
        const H = H_est + k_sys * Q2;
        if (Q > Qmax * 1.5) return null;
        return { Q, H };
    }

    // Intersección Simple (caso N=1)
    function intersectSimple(H_est, k_sys, H0, k_pump, Qmax) {
        return intersectParN(H_est, k_sys, H0, k_pump, 1, Qmax);
    }

    /* ─── ACTUALIZACIÓN Y RENDERIZADO ─── */
    function updateChart() {
        // 1. Obtener Inputs
        const p = {
            H_g: getVal('H_g'),
            P_res: getVal('P_res'),
            H_f: getVal('H_f'),
            Q_dis: getVal('Q_dis'),
            H0: getVal('H0'),
            H_nom: getVal('H_nom'),
            Q_nom: getVal('Q_nom'),
            Q_max: getVal('Q_max')
        };

        const show = {
            sys: document.getElementById('show_sys').checked,
            simple: document.getElementById('show_simple').checked,
            parallel: document.getElementById('show_parallel').checked,
            series: document.getElementById('show_series').checked,
        };

        // 2. Calcular Constantes
        const H_est = p.H_g + p.P_res;
        const k_sys = calcK_sistema(p.H_f, p.Q_dis);
        const k_pump = calcK_bomba(p.H0, p.H_nom, p.Q_nom);
        
        // Calcular Puntos de Trabajo
        const ptS = intersectSimple(H_est, k_sys, p.H0, k_pump, p.Q_max);
        const ptP = intersectParN(H_est, k_sys, p.H0, k_pump, state.n_paralelo, p.Q_max);
        const ptSer = intersectSerN(H_est, k_sys, p.H0, k_pump, state.n_serie, p.Q_max);

        // 3. Generar Datos (Arrays)
        const steps = 100;
        const labels = [];
        const d_sys=[], d_simple=[], d_par=[], d_ser=[];

        for (let i = 0; i <= steps; i++) {
            const Q = (i / steps) * p.Q_max;
            labels.push(Q.toFixed(1));
            const Q2 = Q*Q;

            // Sistema
            if (show.sys) d_sys.push(H_est + k_sys * Q2);

            // Simple
            if (show.simple) {
                const h = p.H0 - k_pump * Q2;
                d_simple.push(h > 0 ? h : null);
            }

            // Paralelo N
            if (show.parallel) {
                // H = H0 - k * (Q/N)^2
                const h = p.H0 - k_pump * ((Q/state.n_paralelo)**2);
                d_par.push(h > 0 ? h : null);
            }

            // Serie N
            if (show.series) {
                // H = N * (H0 - k * Q^2)
                const h_single = p.H0 - k_pump * Q2;
                const h = h_single * state.n_serie;
                d_ser.push(h_single > 0 ? h : null);
            }
        }

        // 4. Actualizar Resultados HTML (Badges)
        const resBox = document.getElementById('resultsContainer');
        resBox.innerHTML = '';

        if(show.sys) {
            const H_design = H_est + k_sys * (p.Q_dis**2);
            resBox.innerHTML += `
            <div class="badge sys">
                <h4>Requerimiento (Diseño)</h4>
                <div class="value">${p.Q_dis.toFixed(1)} m³/h</div>
                <div class="value" style="font-size:0.9em;color:#666">${H_design.toFixed(2)} m.c.a.</div>
                <div class="sub">H_geo: ${H_est.toFixed(2)}m</div>
            </div>`;
        }
        
        if (show.simple && ptS) {
            resBox.innerHTML += `
            <div class="badge simple">
                <h4>1 Bomba</h4>
                <div class="value">${ptS.Q.toFixed(2)} m³/h</div>
                <div class="value" style="font-size:0.9em;color:#666">${ptS.H.toFixed(2)} m.c.a.</div>
            </div>`;
        }

        if (show.parallel && ptP) {
            resBox.innerHTML += `
            <div class="badge parallel">
                <h4>${state.n_paralelo} en Paralelo</h4>
                <div class="value">${ptP.Q.toFixed(2)} m³/h</div>
                <div class="value" style="font-size:0.9em;color:#666">${ptP.H.toFixed(2)} m.c.a.</div>
                ${state.n_paralelo > 1 ? `<div class="sub">C/u: ${(ptP.Q/state.n_paralelo).toFixed(1)} m³/h</div>` : ''}
            </div>`;
        }

        if (show.series && ptSer) {
            resBox.innerHTML += `
            <div class="badge series">
                <h4>${state.n_serie} en Serie</h4>
                <div class="value">${ptSer.Q.toFixed(2)} m³/h</div>
                <div class="value" style="font-size:0.9em;color:#666">${ptSer.H.toFixed(2)} m.c.a.</div>
                ${state.n_serie > 1 ? `<div class="sub">C/u: ${(ptSer.H/state.n_serie).toFixed(1)} mca</div>` : ''}
            </div>`;
        }

        // 5. Actualizar Fórmulas Footer
        const formulasDiv = document.getElementById('formulasContainer');
        formulasDiv.innerHTML = `
            <div class="formula-tag" style="color:${C.sys};border-color:${C.sys}">Sys: H = ${(p.H_g+p.P_res).toFixed(2)} + ${k_sys.toFixed(4)}·Q²</div>
            <div class="formula-tag" style="color:${C.simple};border-color:${C.simple}">1B: H = ${p.H0} - ${k_pump.toFixed(4)}·Q²</div>
            <div class="formula-tag" style="color:${C.par};border-color:${C.par}">Par(${state.n_paralelo}): H = ${p.H0} - ${k_pump.toFixed(4)}·(Q/${state.n_paralelo})²</div>
            <div class="formula-tag" style="color:${C.ser};border-color:${C.ser}">Ser(${state.n_serie}): H = ${state.n_serie}·(${p.H0} - ${k_pump.toFixed(4)}·Q²)</div>
        `;


        // 6. Chart.js Config
        const datasets = [];

        // Puntos de trabajo (Scatter)
        const commonPoint = { radius: 6, hoverRadius: 8, borderWidth: 2, borderColor: '#fff' };

        if(show.sys) {
             // Punto de Diseño
             datasets.push({
                type: 'scatter', label: 'Punto Diseño',
                data: [{x: p.Q_dis, y: H_est + k_sys * (p.Q_dis**2)}],
                backgroundColor: 'white', borderColor: C.sys, borderWidth:2, radius: 5
            });
            // Línea Sistema
            datasets.push({
                type: 'line', label: 'Curva Sistema', data: d_sys,
                borderColor: C.sys, borderDash: [5,5], borderWidth: 2, pointRadius: 0, fill: false, tension:0.4
            });
        }

        if(show.simple) {
            if(ptS) datasets.push({ type: 'scatter', label: 'Op 1 Bomba', data: [{x:ptS.Q, y:ptS.H}], backgroundColor: C.simple, ...commonPoint });
            datasets.push({ type: 'line', label: 'Curva 1 Bomba', data: d_simple, borderColor: C.simple, borderWidth: 2, pointRadius: 0, tension:0.4 });
        }

        if(show.parallel) {
            if(ptP) datasets.push({ type: 'scatter', label: `Op ${state.n_paralelo} Paralelo`, data: [{x:ptP.Q, y:ptP.H}], backgroundColor: C.par, ...commonPoint });
            datasets.push({ type: 'line', label: `Curva ${state.n_paralelo} Paralelo`, data: d_par, borderColor: C.par, borderWidth: 2, pointRadius: 0, tension:0.4 });
        }

        if(show.series) {
            if(ptSer) datasets.push({ type: 'scatter', label: `Op ${state.n_serie} Serie`, data: [{x:ptSer.Q, y:ptSer.H}], backgroundColor: C.ser, ...commonPoint });
            datasets.push({ type: 'line', label: `Curva ${state.n_serie} Serie`, data: d_ser, borderColor: C.ser, borderWidth: 2, pointRadius: 0, tension:0.4 });
        }

        // Render
        const ctx = document.getElementById('pumpChart').getContext('2d');
        if (chartInstance) chartInstance.destroy();

        chartInstance = new Chart(ctx, {
            data: { labels: labels, datasets: datasets },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                interaction: { mode: 'index', intersect: false },
                plugins: {
                    legend: { position: 'bottom', labels: { usePointStyle: true, boxWidth: 8, padding: 20 } },
                    tooltip: {
                        backgroundColor: 'rgba(0, 48, 135, 0.95)',
                        titleColor: '#fff',
                        bodyColor: '#fff',
                        callbacks: {
                            label: (ctx) => {
                                let val = ctx.raw.y !== undefined ? ctx.raw.y : ctx.raw;
                                return `${ctx.dataset.label}: ${val.toFixed(2)} m.c.a.`;
                            }
                        }
                    }
                },
                scales: {
                    x: {
                        title: { display: true, text: 'Caudal Q (m³/h)', color: '#64748b' },
                        grid: { color: '#f1f5f9' },
                        ticks: { maxTicksLimit: 10 }
                    },
                    y: {
                        title: { display: true, text: 'Altura H (m.c.a.)', color: '#64748b' },
                        grid: { color: '#f1f5f9' },
                        min: 0
                    }
                }
            }
        });
    }

    // Inicializar
    window.onload = updateChart;

</script>
